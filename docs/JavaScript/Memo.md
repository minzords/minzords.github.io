# Les variables
```js
	// Déclaration de la variable a
	let a = 1
  // Réaffectation de la variable
  a = 66
```

# Les contantes
```js
	const NOM = "Le Nom"
```

# Utilisation de l'objet
```js
  let distribution = {
  nom: "ArchLinux",
  version: "rolling",
  kernel: 5.1,
  }
```

```js
	distribution.gespaquet = "pacman"
```
